          if (window.top == window.self) {

          } else {
            normalizedLink = document.createElement('a');
            normalizedLink.href = "https://custom-pins.myshopify.com/admin/oauth/authorize?client_id=abcdef&scope=write_products%2Cread_orders&redirect_uri=https%3A%2F%2Fexample.com%2Foauth%2Fcallback";

              message: 'Shopify.API.remoteRedirect',

            });
